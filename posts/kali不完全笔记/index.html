Kali2020.4#

设置root密码#

sudo passwd root

升级更新#

配置更新源#

vi /etc/apt/sources.list
deb http://http.kali.org/kali kali-rolling main non-free contrib
deb http://mirrors.aliyun.com/debian stable main contrib non-free
deb http://mirrors.ustc.edu.cn/kali kali-rolling main non-free contrib

更新软件及系统#

apt update && apt upgrade -y   //升级软件列表并更新软件未命名.png
apt dist-upgrade -y //升级系统

安装内核头文件#

apt install linux-headers-$(uname -r)

清理系统#

apt clean && apt autoclean && apt autoremove -y
apt remove XXX   //删除软件，但保留配置
apt purge XXX    //删除软件，删除配置
apt clean        //删除软件包缓存(/var/cache/apt/archives)
apt autoremove   //自动清理无用的依赖包
dpkg -l |grep ^rc|awk '{print $2}' |sudo xargs dpkg -P   //清理系统残存配置

安装软件#

安装中文输入法#

apt-get update
apt-get install ibus ibus-pinyin
im-config   //选择ibus输入法
cd~
vi .bashrc

输入以下字符：

reboot重启即可。

安装docker#

apt install -y docker.io
apt-get install docker-compose
//配置docker国内镜像源
vi /etc/docker/daemon.json
systemctl start docker.service
systemctl enable docker.service

/etc/docker/daemon.json:

{
  "registry-mirrors" : [
    "http://ovfftd6p.mirror.aliyuncs.com",
    "http://registry.docker-cn.com",
    "http://docker.mirrors.ustc.edu.cn",
    "http://hub-mirror.c.163.com"
  ],
  "insecure-registries" : [
    "registry.docker-cn.com",
    "docker.mirrors.ustc.edu.cn"
  ],
  "debug" : true,
  "experimental" : true
}

安装AWVS13#

docker pull secfa/docker-awvs       //pull拉取awvs13下载镜像
docker run --name awvs13 -it -d -p 13443:3443 secfa/docker-awvs //将docker的3443端口映射到物理机的13443端口
//awvs13 info
awvs13 username: admin@admin.com
awvs13 password: Admin123
AWVS版本：13.0.200217097

安装nessus#

docker pull takaharuogasa/nessus2docker
docker run -d -it --name nessus -p 8834:8834 takaharuogasa/nessus2docker:latest
激活码：K5X9-FFTP-4CDT-W4SK

或者直接安装awvs13和nessus

# 拉取镜像
docker pull leishianquan/awvs-nessus:v2
# 启动
docker run --name awvs-nessus -it -d -p 13443:3443 -p 8834:8834 leishianquan/awvs-nessus:v2
# 查看容器
docker ps –a
# 启动容器
docker start container-id
# 进入容器
docker exec –it container-id /bin/bash

# 进入容器后，启动nessus
/etc/init.d/nessusd start

# 访问扫描器地址和账号密码
